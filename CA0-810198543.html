def htprd(t1 , t0, x):
    out = [t1*i + t0 for i in x]
    return out

def mse(yi, ht):
    n = len(yi)
    pw = sum(pow((yi - ht), 2))
    s = pw / n
    return s

def slin(data):

    C=[]
    C = np.full((len(data['FUELCONSUMPTION_COMB_MPG']),1),1)

    X = np.column_stack((C,data['FUELCONSUMPTION_COMB_MPG']))
    Xt = X.transpose()
          
    xtx = np.dot(Xt, X)
    xtxi = np.linalg.inv(xtx)

    xty = np.dot(Xt,  data['CO2EMISSIONS'])

    cal = np.dot(xtxi, xty)
    return cal
